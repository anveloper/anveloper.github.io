## 문제

[BOJ 3232 - Multiply](https://www.acmicpc.net/problem/3232)

## 풀이

문제의 조건을 그대로 구현한다.

## 코드

```cpp
#include <iostream>
#include <string>
#include <cmath>

using namespace std;

int multiply(int num, int x)
{
  int sum = 0;
  int power = 1;
  string number = to_string(num);
  for (int j = number.length() - 1; j >= 0; j--)
  {
    sum += (number[j] - '0') * power;
    power *= x;
  }
  return sum;
}

bool isLarge(int num, int x)
{
  string str = to_string(num);
  for (auto c : str)
  {
    if (c - '0' >= x)
      return false;
  }
  return true;
}
int main()
{
  int T;
  cin >> T;
  for (int i = 0; i < T; i++)
  {
    int p, q, r;
    cin >> p >> q >> r;
    for (int B = 2; B < 17; B++)
    {
      if (!isLarge(p, B) || !isLarge(q, B) || !isLarge(r, B))
        continue;

      if (multiply(p, B) * multiply(q, B) == multiply(r, B))
      {
        cout << B << "\n";
        break;
      }
      else if (B == 16)
        cout << "0" << "\n";
    }
  }
}
```

## 복잡도

- 시간: O(N)
- 공간: O(N)
